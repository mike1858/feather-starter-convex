    **Step 1: Create convex/devErrors/actions.ts** (per D-08, D-09, D-11):

    ```typescript
    import { internalAction } from "@cvx/_generated/server";
    import { internal } from "@cvx/_generated/api";

    export const sendDigest = internalAction({
      args: {},
      handler: async (ctx) => {
        // 1. Get undigested errors
        const errors = await ctx.runQuery(internal.devErrors.queries.listUndigested);

        if (errors.length === 0) return;

        // 2. Build digest (per D-11)
        const errorCount = errors.length;
        const topErrors = buildTopErrors(errors, 10); // top 10 by frequency
        const digest = {
          errorCount,
          topErrors,
          environment: {
            timestamp: Date.now(),
            source: "feather-client",
          },
          // No user PII, no data values (per D-11)
        };

        // 3. POST to configured endpoint
        const endpoint = process.env.ERROR_DIGEST_ENDPOINT;
        const secret = process.env.ERROR_DIGEST_SECRET;

        if (endpoint && secret) {
          try {
            await fetch(endpoint, {
              method: "POST",
              headers: {
                "Content-Type": "application/json",
                "Authorization": `Bearer ${secret}`,
              },
              body: JSON.stringify(digest),
            });
          } catch (e) {
            // Fire-and-forget: log but don't fail
            console.error("[digest] Failed to POST:", e);
          }
        }

        // 4. Mark as digested
        const ids = errors.map((e: any) => e._id);
        // Batch in groups of 50 for Convex arg size limits
        for (let i = 0; i < ids.length; i += 50) {
          await ctx.runMutation(internal.devErrors.mutations.markDigestedInternal, {
            ids: ids.slice(i, i + 50),
          });
        }
      },
    });

    function buildTopErrors(errors: any[], limit: number) {
      const freq = new Map<string, { message: string; stack?: string; functionName?: string; count: number }>();
      for (const err of errors) {
        const key = err.message;
        const existing = freq.get(key);
        if (existing) {
          existing.count++;
        } else {
          freq.set(key, {
            message: err.message,
            stack: err.stack,
            functionName: err.functionName,
            count: 1,
          });
        }
      }
      return Array.from(freq.values())
        .sort((a, b) => b.count - a.count)
        .slice(0, limit);
    }
    ```

    ```typescript
    import { httpRouter } from "convex/server";
    import { httpAction } from "./_generated/server";
    import { internal } from "./_generated/api";
    import { auth } from "./auth";

    const http = httpRouter();
    auth.addHttpRoutes(http);

    // Error digest receiving endpoint (for admin Convex instances)
    const receiveDigest = httpAction(async (ctx, request) => {
      const authHeader = request.headers.get("Authorization");
      const expectedSecret = process.env.DIGEST_RECEIVE_SECRET;

      if (!expectedSecret || authHeader !== `Bearer ${expectedSecret}`) {
        return new Response("Unauthorized", { status: 401 });
      }

      const digest = await request.json();

      await ctx.runMutation(internal.devErrors.mutations.storeDigest, {
        digest: JSON.stringify(digest),
        receivedAt: Date.now(),
      });

      return new Response(null, { status: 200 });
    });

    http.route({
      path: "/api/errors/digest",
      method: "POST",
      handler: receiveDigest,
    });

    export default http;
    ```
