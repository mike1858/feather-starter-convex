    **1. `generators/utils/smart-add.js` — Smart file creation with marked region support**

    Export `registerSmartAdd(plop)` that registers a custom action type `smartAdd`:

    ```javascript
    export function registerSmartAdd(plop) {
      plop.setActionType("smartAdd", async (answers, config, plop) => {
        const targetPath = plop.renderString(config.path, answers);
        const templateContent = await renderTemplate(config.templateFile, answers, plop);

        if (fs.existsSync(targetPath)) {
          // Re-run mode: extract custom regions from existing file
          const existing = fs.readFileSync(targetPath, "utf8");
          const customRegions = extractCustomRegions(existing);
          const merged = injectCustomRegions(templateContent, customRegions);
          fs.writeFileSync(targetPath, merged);
          return `Updated ${targetPath} (preserved ${customRegions.size} custom regions)`;
        } else {
          // First run: write template as-is
          ensureDir(path.dirname(targetPath));
          fs.writeFileSync(targetPath, templateContent);
          return `Created ${targetPath}`;
        }
      });
    }
    ```

    Helper `renderTemplate` uses Plop's Handlebars instance to render a template file against the answers object. Read the `.hbs` file, compile with Handlebars, render with `answers` (which includes the full resolved YAML config).

    Import `extractCustomRegions` and `injectCustomRegions` from `./marked-regions.js` — these are plain ESM JavaScript files created in Plan 01, directly importable with no compilation step.
